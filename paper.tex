% 新建一个计数器来控制 interlude 的编号
\newcounter{interludecounter}
\setcounter{interludecounter}{0} 

% 定义 \interlude，显示为 "幕间 序号 标题"
\newcommand{\interlude}[1]{ 
  \stepcounter{interludecounter} 
  \chapter*{幕间 \theinterludecounter\\[25pt] #1} 
  \addcontentsline{toc}{chapter}{幕间 \theinterludecounter\ #1} 
  \markboth{Interlude}{幕间} 
}

  \addcontentsline{toc}{chapter}{导言\ #1} 
  \markboth{Introduction}{导言} % 设置页眉
}
